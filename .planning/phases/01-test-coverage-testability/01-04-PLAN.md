    Create `.github/workflows/ci.yml`:

    ```yaml
    name: CI

    on:
      push:
        branches: [main]
      pull_request:
        branches: [main]

    jobs:
      lint-and-test:
        runs-on: ubuntu-latest
        steps:
          - name: Checkout
            uses: actions/checkout@v4

          - name: Setup Node.js
            uses: actions/setup-node@v4
            with:
              node-version: 22
              cache: npm

          - name: Install dependencies
            run: npm ci

          - name: Type check
            run: npm run check

          - name: Unit tests
            run: npm test

          - name: Build (required for E2E)
            run: npm run build

          - name: Install Playwright browsers
            run: npx playwright install --with-deps chromium

          - name: E2E tests
            run: npm run test:e2e
    ```

    Key decisions:
    - Node 22 matches `deploy.yml` (line `node-version: 22`)
    - `actions/checkout@v4` and `actions/setup-node@v4` match deploy workflow versions
    - Type check runs before tests (faster feedback on type errors)
    - Unit tests run before build (they don't need built output)
    - Build must happen before E2E tests (Playwright preview server serves `dist/`)
    - `npx playwright install --with-deps chromium` installs browser + system deps in CI
    - Runs on both push to main AND pull requests (deploy.yml only runs on push to main)
  </action>

  <action>
    The deploy workflow should depend on the CI workflow passing. Since CI is a separate workflow file, we cannot use `needs:` across workflow files directly. Instead, we have two options:

    **Option A (recommended): Use `workflow_run` trigger on deploy.yml.**
    This makes deploy.yml run AFTER ci.yml completes successfully on main.

    Update `deploy.yml`:

    ```yaml
    name: Deploy to GitHub Pages

    on:
      workflow_run:
        workflows: ["CI"]
        branches: [main]
        types: [completed]
      workflow_dispatch:

    permissions:
      contents: read
      pages: write
      id-token: write

    concurrency:
      group: pages
      cancel-in-progress: false

    jobs:
      build:
        runs-on: ubuntu-latest
        if: ${{ github.event_name == 'workflow_dispatch' || github.event.workflow_run.conclusion == 'success' }}
        steps:
          - name: Checkout
            uses: actions/checkout@v4

          - name: Setup Node.js
            uses: actions/setup-node@v4
            with:
              node-version: 22
              cache: npm

          - name: Install dependencies
            run: npm ci

          - name: Build
            run: npm run build

          - name: Upload artifact
            uses: actions/upload-pages-artifact@v3
            with:
              path: dist/

      deploy:
        needs: build
        runs-on: ubuntu-latest
        environment:
          name: github-pages
          url: ${{ steps.deployment.outputs.page_url }}
        steps:
          - name: Deploy to GitHub Pages
            id: deployment
            uses: actions/deploy-pages@v4
    ```

    Key changes from current:
    1. `on.push` replaced with `on.workflow_run` referencing the "CI" workflow
    2. Added `if` condition on `build` job: only runs if CI succeeded OR manual dispatch
    3. `workflow_dispatch` kept for manual deploys
    4. Everything else (permissions, concurrency, steps) remains identical

    This ensures: push to main → CI runs → if CI passes → deploy runs. If CI fails, deploy is skipped.
  </action>
